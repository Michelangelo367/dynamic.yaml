files:
  - name:     alpha
    outpath:  root/txt/alpha.txt
    wwbody: |
      Hello world!
      This is the content in alpha.txt

  - name:     bravo
    outpath:  root/txt/bravo.txt
    wwbody: |
      Hello world!
      This is the content in bravo.txt

  - name:     charlie
    outpath:  root/txt/charlie.txt
    wwbody: |
      Hello world!
      This is the content in charlie.txt

__yaml__:
  - template: |
      {% for file in files %}
        {%filter jjdedent |trim |jjtozipfile('',file.outpath) %}
          {{ file.wwbody |upper }}
        {%endfilter-%}      
      {% endfor %}
